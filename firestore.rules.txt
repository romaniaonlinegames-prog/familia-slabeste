// Reguli de securitate Firestore pentru "Familia Slăbește"
//
// Deja completate cu email-urile reale ale conturilor voastre.
// Dacă vreodată adăugați un cont nou (alt membru, alt email), adăugați-l
// în lista de mai jos și apăsați din nou "Publish" în Firebase Console
// (Firestore Database → tab "Rules").
//
// Ce fac: permit citire/scriere DOAR celor doi utilizatori autentificați
// cu parolă, ale căror email-uri sunt în lista de mai jos. Chiar dacă
// tot codul aplicației (inclusiv acest fișier) e public pe GitHub,
// nimeni altcineva nu poate intra fără parola unuia dintre voi doi.

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    match /families/{familyId}/data/{document} {
      allow read, write: if request.auth != null
        && request.auth.token.email in [
          "bphoto.ro@gmail.com",
          "marincus.carmen@gmail.com"
        ];
    }
  }
}
