// Reguli de securitate Firestore pentru "Familia Slăbește"
//
// Cum le pui: Firebase Console → Firestore Database → tab "Rules" →
// înlocuiește tot conținutul cu ce e mai jos → Publish.
//
// Model nou: fiecare cont (users/{uid}) e legat de o familie
// (families/{familyId}), prin câmpul "familyId". Doar membrii unei
// familii pot citi/scrie datele acelei familii.

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // Fiecare utilizator își poate citi/scrie DOAR propriul document
    // de cont (acolo ținem legătura cu familia lui).
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }

    // Datele unei familii — permise doar celor al căror cont (users/{uid})
    // are familyId == familyId-ul cerut aici.
    match /families/{familyId}/data/{document} {
      allow read, write: if request.auth != null
        && get(/databases/$(database)/documents/users/$(request.auth.uid)).data.familyId == familyId;
    }
  }
}
